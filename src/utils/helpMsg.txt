Parameters:

    --action=output-file (-a output-file) - output data from file to console

    --action=convert-from-file (-a convert-from-file) - convert file from CSV to JSON and output to console

    --action=convert-to-file (-a convert-to-file) - convert file from CSV to JSON and output to file
